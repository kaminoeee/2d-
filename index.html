// APIキーは不要です！代わりに公開されているInvidiousのインスタンスURLを使用します const BASE_URL = 'https://invidious.perennialte.ch/api/v1'; // 調子が悪い時は別のインスタンスURLに変更可能 window.onload = () => { fetchTrendingVideos(); }; // トレンド動画の取得 async function fetchTrendingVideos() { // Invidiousのトレンド取得API const url = `${BASE_URL}/trending?region=JP`; try { const response = await fetch(url); const data = await response.json(); // Invidiousのデータ構造に合わせてマッピング displayVideos(data); } catch (error) { console.error('エラーが発生しました:', error); } } // 動画検索 async function searchVideos() { const query = document.getElementById('search-input').value; if (!query) return; // Invidiousの検索API const url = `${BASE_URL}/search?q=${encodeURIComponent(query)}&type=video`; try { const response = await fetch(url); const data = await response.json(); displayVideos(data); } catch (error) { console.error('エラーが発生しました:', error); } } // 動画一覧を画面に描画 function displayVideos(videos) { const grid = document.getElementById('video-grid'); grid.innerHTML = ''; videos.forEach(video => { const card = document.createElement('div'); card.className = 'video-card'; card.onclick = () => playVideo(video.videoId, video.title); // サムネイルもInvidiousがキャッシュしている画像などを利用可能 const thumbnail = video.videoThumbnails ? video.videoThumbnails.find(t => t.quality === 'medium').url : ''; card.innerHTML = `

${video.title}

${video.author}

`; grid.appendChild(card); }); } // 動画を再生（再生はYouTube公式の埋め込みでも、Invidiousの再生ソースでもOK） function playVideo(videoId, title) { const playerSection = document.getElementById('player-section'); const player = document.getElementById('youtube-player'); const titleElement = document.getElementById('video-title'); // 広告を減らしたい場合は公式埋め込みではなくInvidiousの埋め込みURL（例: https://invidious.perennialte.ch/embed/${videoId}）にすることも可能 player.src = `https://www.youtube.com/embed/${videoId}?autoplay=1`; titleElement.textContent = title; playerSection.style.display = 'block'; playerSection.scrollIntoView({ behavior: 'smooth' }); }
